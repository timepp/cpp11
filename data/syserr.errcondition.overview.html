[syserr.errcondition]19.5.3 Class error_condition

[syserr.errcondition.overview]19.5.3.1 Class error_condition overview

1The class error_condition describes an object used to hold values identifying error conditions. [ Note: error_condition values are portable abstractions, while error_code values (19.5.2) are implementation specific. — end note ]

namespace std {
  class error_condition {
  public:
    // 19.5.3.2 constructors:
    error_condition() noexcept;
    error_condition(int val, const error_category& cat) noexcept;
    template <class ErrorConditionEnum>
      error_condition(ErrorConditionEnum e) noexcept;
        
    // 19.5.3.3 modifiers:
    void assign(int val, const error_category& cat) noexcept;
    template<class ErrorConditionEnum>
      error_condition& operator=(ErrorConditionEnum e) noexcept;
    void clear() noexcept;
    
    // 19.5.3.4 observers:
    int value() const noexcept;
    const error_category& category() const noexcept;
    string message() const;
    explicit operator bool() const noexcept;
     
  private:
    int val_;                   // exposition only
    const error_category* cat_; // exposition only
  };

  // 19.5.3.5 non-member functions:
  bool operator<(const error_condition& lhs, const error_condition& rhs) noexcept;
} // namespace std
